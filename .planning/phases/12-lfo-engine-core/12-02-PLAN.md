```cpp
TEST_CASE("LfoEngine free mode accumulates phase without drift", "[lfo][free][perf03]")
{
    LfoEngine lfo;
    auto p = baseParams();
    p.syncMode     = false;
    p.isDawPlaying = false;
    p.rateHz       = 1.0f;       // 1 Hz = 1 cycle per second
    p.sampleRate   = 1000.0;     // 1000 samples per second → 1 cycle = 1000 samples
    p.blockSize    = 100;         // 10 blocks per cycle
    p.waveform     = LfoEngine::Waveform::SawUp;
    p.level        = 1.0f;

    // After 5 blocks (500 samples), phase should be 0.5 → SawUp output = 0.0
    for (int i = 0; i < 5; ++i)
        lfo.process(p);

    // 5 blocks of 100 samples at 1Hz/1000Hz = 5 * 0.1 = 0.5 phase
    // SawUp at phi=0.5 → 2*0.5-1 = 0.0
    const float out = lfo.process(p);  // 6th block: phase 0.5..0.6, returns at 0.5
    // output should be near 0.0 (SawUp at ~0.5 phase)
    // Note: process() returns value at start of block's phase; tolerance is generous
    CHECK(std::abs(out) < 0.2f);  // within 20% of 0
}
```
