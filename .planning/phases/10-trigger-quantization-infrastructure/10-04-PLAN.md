    ```cpp
    // ── Quantize mode buttons (Off / Live / Post) ─────────────────────────────
    constexpr int kQuantizeRadioGroup = 1;  // No other radio groups in this editor (confirmed by grep)

    quantizeOffBtn_ .setButtonText("Off");
    quantizeLiveBtn_.setButtonText("Live");
    quantizePostBtn_.setButtonText("Post");

    for (auto* btn : { &quantizeOffBtn_, &quantizeLiveBtn_, &quantizePostBtn_ })
    {
        btn->setRadioGroupId(kQuantizeRadioGroup, juce::dontSendNotification);
        btn->setClickingTogglesState(true);
        styleButton(*btn, false);  // use existing styleButton helper used for all other looper buttons
        addAndMakeVisible(*btn);
    }

    // onClick handlers: write new mode to APVTS via processor
    quantizeOffBtn_.onClick  = [this] {
        proc_.setQuantizeMode(0);
        proc_.looperRevertQuantize();  // switching to Off also reverts any active post-quantize
    };
    quantizeLiveBtn_.onClick = [this] { proc_.setQuantizeMode(1); };
    quantizePostBtn_.onClick = [this] {
        proc_.setQuantizeMode(2);
        proc_.looperApplyQuantize();  // switching to Post immediately applies quantize
    };

    // Restore button state from APVTS on load (APVTS may have restored a non-zero mode)
    {
        const int savedMode = proc_.getQuantizeMode();
        quantizeOffBtn_ .setToggleState(savedMode == 0, juce::dontSendNotification);
        quantizeLiveBtn_.setToggleState(savedMode == 1, juce::dontSendNotification);
        quantizePostBtn_.setToggleState(savedMode == 2, juce::dontSendNotification);
    }

    // ── Quantize subdivision dropdown ─────────────────────────────────────────
    quantizeSubdivBox_.addItem("1/4",  1);  // APVTS int 0 → ComboBox ID 1 (JUCE offset convention)
    quantizeSubdivBox_.addItem("1/8",  2);  // APVTS int 1 → ComboBox ID 2
    quantizeSubdivBox_.addItem("1/16", 3);  // APVTS int 2 → ComboBox ID 3
    quantizeSubdivBox_.addItem("1/32", 4);  // APVTS int 3 → ComboBox ID 4
    addAndMakeVisible(quantizeSubdivBox_);

    // ComboBoxParameterAttachment: APVTS quantizeSubdiv (int 0..3) ↔ ComboBox items 1..4
    // This is the same pattern as loopSubdivAtt_, randomSubdivAtt_ — fully handles persistence.
    quantizeSubdivAtt_ = std::make_unique<ComboAtt>(
        proc_.apvts, "quantizeSubdiv", quantizeSubdivBox_);
    ```
