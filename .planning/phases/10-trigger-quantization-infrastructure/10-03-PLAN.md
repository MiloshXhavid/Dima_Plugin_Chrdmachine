    ```cpp
    void LooperEngine::applyQuantizeToStore(double gridSize, double loopLen)
    {
        const int n = playbackCount_.load(std::memory_order_relaxed);
        if (n == 0) return;

        // Step 1: snap all Gate events; leave Joystick/Filter events untouched.
        // Track per-voice last snapped gate-on position for duration-preserving note-off.
        double lastOnBeat[4] = { -1.0, -1.0, -1.0, -1.0 };  // -1 = no pending note-on

        for (int i = 0; i < n; ++i)
        {
            auto& ev = playbackStore_[i];
            if (ev.type != LooperEvent::Type::Gate) continue;

            const bool isOn = (ev.value > 0.5f);

            if (isOn)
            {
                const double snapped = snapToGrid(ev.beatPosition, gridSize, loopLen);
                lastOnBeat[ev.voice] = ev.beatPosition;  // save original on-beat for duration calc
                ev.beatPosition = snapped;
            }
            else
            {
                // Duration-preserving shift for note-off
                if (lastOnBeat[ev.voice] >= 0.0)
                {
                    double duration = ev.beatPosition - lastOnBeat[ev.voice];
                    if (duration < 0.0) duration += loopLen;  // wrap case
                    const double minGate = 1.0 / 64.0;
                    duration = std::max(duration, minGate);
                    // snapped on-beat was already stored (previous Gate-on iteration set ev.beatPosition)
                    // We need the snapped-on position. Find it by re-snapping the saved original:
                    const double snappedOn = snapToGrid(lastOnBeat[ev.voice], gridSize, loopLen);
                    ev.beatPosition = std::fmod(snappedOn + duration, loopLen);
                    lastOnBeat[ev.voice] = -1.0;  // reset tracking
                }
                // If no matching note-on tracked, leave note-off beat unchanged
            }
        }

        // Step 2: sort by beatPosition (matches finaliseRecording sort behavior)
        std::sort(playbackStore_.begin(), playbackStore_.begin() + n,
                  [](const LooperEvent& a, const LooperEvent& b) {
                      return a.beatPosition < b.beatPosition;
                  });

        // Step 3: deduplicate — same voice + same snapped beat Gate-on → keep first, discard later
        // Mark duplicates as Gate-off with beatPosition = -1 (sentinel for removal)
        bool seenOnBeat[4][2] = {};  // [voice][0=just a flag placeholder] — use a local set approach
        // Simpler: scan for Gate-on events with beatPosition within epsilon of each other, same voice.
        // After sort, duplicates are adjacent. Compare consecutive Gate-on events for same voice.
        int writeIdx = 0;
        int lastOnIdx[4] = { -1, -1, -1, -1 };

        // Re-pack to remove duplicates in-place using a second pass
        std::array<LooperEvent, LOOPER_FIFO_CAPACITY> deduped {};
        writeIdx = 0;
        for (int i = 0; i < n; ++i)
        {
            const auto& ev = playbackStore_[i];
            if (ev.type == LooperEvent::Type::Gate && ev.value > 0.5f)
            {
                // Check if we already have a gate-on for this voice at this beat position
                bool isDuplicate = false;
                for (int j = 0; j < writeIdx; ++j)
                {
                    const auto& prev = deduped[j];
                    if (prev.type == LooperEvent::Type::Gate && prev.value > 0.5f
                        && prev.voice == ev.voice
                        && std::abs(prev.beatPosition - ev.beatPosition) < 1e-9)
                    {
                        isDuplicate = true;
                        break;
                    }
                }
                if (isDuplicate) continue;
            }
            if (writeIdx < LOOPER_FIFO_CAPACITY)
                deduped[writeIdx++] = ev;
        }

        // Copy deduped back to playbackStore_
        for (int i = 0; i < writeIdx; ++i)
            playbackStore_[i] = deduped[i];
        playbackCount_.store(writeIdx, std::memory_order_release);
    }
    ```
